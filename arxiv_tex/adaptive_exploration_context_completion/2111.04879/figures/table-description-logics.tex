\newcommand\setarraystretch[1]{%
  \noalign{\ifnum0=`}\fi
  \gdef\arraystretch{#1}%
  \global\setbox\@arstrutbox\hbox{%
    \vrule \@height\arraystretch\ht\strutbox%
    \@depth\arraystretch \dp\strutbox%
    \@width\z@}%
  \ifnum0=`{\fi}%
}
