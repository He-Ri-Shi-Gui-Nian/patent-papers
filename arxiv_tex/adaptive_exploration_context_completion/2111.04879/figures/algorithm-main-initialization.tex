\begin{algorithm}
	\DontPrintSemicolon
	\caption{Initialization of the population via biased random walks.}%
	\label{alg:main-initialization}
    \textbf{Input:} Knowledge base $K$, Positive examples $E^{+}$, Max.\ adjacent roles $\mathit{maxT}$, List of $splits$ for each data property $p$\\
    	\textbf{Output:} Set of concepts $population$ \\
    	\SetKwFunction{FMain}{GetConceptsFromExamples}
    	\SetKwProg{Pn}{Function}{:}{\KwRet $population$}
    	\Pn{\FMain{$K$, $E^+$, $maxT$, $splits$}}{
    		Compute the frequencies $\mathit{ft}[c]$ of the classes $c$ of the positive examples $E^{+}$\;
    		$population = \emptyset$\;
    		\While{$len(population) < populationSize$}{
    		Uniformly randomly pick $e^{+} \in E^{+}$\;
    		$population = population \cup \{$\texttt{GetConceptFromExample}${(K, e^{+}, \mathit{ft}, \mathit{maxT}, \mathit{splits})}\}$\;		
    		}
    	}
\end{algorithm}
