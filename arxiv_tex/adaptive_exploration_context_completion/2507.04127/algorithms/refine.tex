\begin{algorithm}[t]
\centering
\caption{\byokg Framework.} \label{alg:refine}
\begin{algorithmic}[1]
   \STATE {\bfseries Input:} Prompt $p$, Query  $q$, Schema $S$, Iterations $T_R$
   \STATE $\gC \gets \emptyset$
   \FORALL{$t \in T_R$} 
   \STATE $\tilde{\gE}, \tilde{\gP}, \tilde{\gQ}, \tilde{\gA} = $ LLM($p$, $q$, $S$, $\gC$)
   \STATE $\gE, \gA = $ EntityLink($\tilde{\gE}, \tilde{\gA}$)
   \STATE {\bfseries if} $\tilde{\gE} = \emptyset$: {\bfseries break}
   \STATE $\gP = $ PathRetrieve($\tilde{\gP}, \tilde{\gE}, \tilde{\gA}$)
   \STATE $\gQ_a$ = QueryRetrieve($\tilde{\gQ}$)
   \STATE $\gT_q$ = TripletRetrieve($q, \gE \cup \gA, \gC$)
   \STATE $\gC \gets \gT_q \cup \gP \cup \gQ_a \cup \gC$
    
   \ENDFOR
   \STATE  $\gA  =$ LLM($q, \gC$).
   \STATE {\bfseries Output:} Answers $\gA$.
\end{algorithmic}
\end{algorithm}
