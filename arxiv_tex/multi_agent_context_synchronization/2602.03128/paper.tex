\newcommand{\accChange}[2]{%
  \pgfmathsetmacro{\diff}{#2 - #1}%
  \pgfmathsetmacro{\len}{max(.1, min(0.05*abs(\diff), .3))}%
  \ifdim \diff pt > 0pt
    #2\,
    \tikz[
      x=1mm,
      >=stealth,
      line width=0.2mm
    ]{
      \draw[->, green!70!black] (1,0) -- ++(0,{0.8*\len});
    }%
  \else
    #2\,
    \tikz[
      x=1mm,
      >=stealth,
      line width=0.2mm
    ]{
      \draw[->, red!70!black] (1,0) -- ++(0,{-0.8*\len});
    }%
  \fi
}
